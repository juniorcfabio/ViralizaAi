// COPIE ESTE CÓDIGO PARA A FUNÇÃO "purchase" NO DASHBOARD SUPABASE
// FUNÇÃO PARA PAGAMENTOS AVULSOS (Ferramentas, Anúncios, Serviços)

import { serve } from "https://deno.land/std@0.168.0/http/server.ts"
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'
import Stripe from 'https://esm.sh/stripe@12.0.0?target=deno'

const stripe = new Stripe(Deno.env.get('STRIPE_SECRET_KEY') || '', {
  apiVersion: '2022-11-15',
})

const supabaseUrl = Deno.env.get('SUPABASE_URL')!
const supabaseServiceKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!

const supabase = createClient(supabaseUrl, supabaseServiceKey)

serve(async (req) => {
  try {
    if (req.method !== 'POST') {
      return new Response('Method not allowed', { status: 405 })
    }

    const { productId, userId, quantity = 1, referralCode } = await req.json()

    if (!productId || !userId) {
      return new Response('Missing required fields', { status: 400 })
    }

    // Buscar produto
    const { data: product, error: productError } = await supabase
      .from('products')
      .select('*')
      .eq('id', productId)
      .eq('is_active', true)
      .single()

    if (productError || !product) {
      return new Response('Product not found', { status: 404 })
    }

    // Buscar usuário
    const { data: user, error: userError } = await supabase
      .from('user_profiles')
      .select('*')
      .eq('id', userId)
      .single()

    if (userError || !user) {
      return new Response('User not found', { status: 404 })
    }

    // Verificar se já tem customer no Stripe
    let customerId: string
    const { data: existingCustomer } = await supabase
      .from('subscriptions')
      .select('stripe_customer_id')
      .eq('user_id', userId)
      .single()

    if (existingCustomer?.stripe_customer_id) {
      customerId = existingCustomer.stripe_customer_id
    } else {
      // Criar customer no Stripe
      const customer = await stripe.customers.create({
        email: user.email,
        name: user.name,
        metadata: {
          user_id: userId,
          supabase_user: 'true'
        }
      })
      customerId = customer.id
    }

    // Processar código de referência se fornecido
    let affiliateId: string | null = null
    let commissionRate = 0
    if (referralCode) {
      const { data: affiliate } = await supabase
        .from('affiliates')
        .select('id, user_id, commission_rate')
        .eq('referral_code', referralCode)
        .eq('is_active', true)
        .single()

      if (affiliate && affiliate.user_id !== userId) {
        affiliateId = affiliate.id
        commissionRate = affiliate.commission_rate || 10
      }
    }

    // Calcular valores
    const unitPrice = product.price
    const totalAmount = unitPrice * quantity
    const commissionAmount = affiliateId ? (totalAmount * commissionRate) / 100 : 0

    // Criar Payment Intent no Stripe
    const paymentIntent = await stripe.paymentIntents.create({
      amount: Math.round(totalAmount * 100), // Stripe usa centavos
      currency: 'brl',
      customer: customerId,
      metadata: {
        user_id: userId,
        product_id: productId,
        product_type: product.type,
        quantity: quantity.toString(),
        ...(affiliateId && { 
          affiliate_id: affiliateId,
          commission_amount: commissionAmount.toString()
        })
      },
      description: `${product.name} - ${user.name}`,
      automatic_payment_methods: {
        enabled: true,
      },
    })

    // Registrar compra no banco
    const { data: purchase, error: purchaseError } = await supabase
      .from('purchases')
      .insert({
        user_id: userId,
        product_id: productId,
        stripe_payment_intent_id: paymentIntent.id,
        stripe_customer_id: customerId,
        amount: totalAmount,
        currency: 'BRL',
        status: 'pending',
        affiliate_id: affiliateId,
        commission_amount: commissionAmount,
        metadata: {
          quantity,
          unit_price: unitPrice,
          product_name: product.name,
          product_type: product.type
        }
      })
      .select()
      .single()

    if (purchaseError) {
      console.error('Erro ao registrar compra:', purchaseError)
      return new Response('Error creating purchase record', { status: 500 })
    }

    // Registrar atividade
    await supabase.from('activity_logs').insert({
      user_id: userId,
      action: 'purchase_initiated',
      details: {
        purchase_id: purchase.id,
        product_id: productId,
        product_name: product.name,
        amount: totalAmount,
        payment_intent_id: paymentIntent.id,
        affiliate_id: affiliateId
      },
      resource_type: 'purchase'
    })

    return new Response(JSON.stringify({
      clientSecret: paymentIntent.client_secret,
      purchaseId: purchase.id,
      amount: totalAmount,
      currency: 'BRL'
    }), {
      status: 200,
      headers: { 'Content-Type': 'application/json' }
    })

  } catch (error) {
    console.error('Erro na compra:', error)
    return new Response(JSON.stringify({
      error: 'Internal server error',
      message: error.message
    }), {
      status: 500,
      headers: { 'Content-Type': 'application/json' }
    })
  }
})
